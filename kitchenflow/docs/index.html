Quick Start

# 1. Start an episode
curl -X POST https://YOUR_HF_USERNAME-kitchenflow-v1.hf.space/reset \
  -H 'Content-Type: application/json' \
  -d '{"task_id": "T1_single_order_dispatch"}'

# 2. Dispatch a driver
curl -X POST https://YOUR_HF_USERNAME-kitchenflow-v1.hf.space/step \
  -H 'Content-Type: application/json' \
  -d '{"action": {"dispatch_decisions": {"ORD001": 1}}, "episode_id": "..."}'
